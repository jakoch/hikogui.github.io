65
66 for (hilet cmd : system) {
67 hilet i = std::find(cache.cbegin(), cache.cend(), cmd);
68 if (i == cache.cend()) {
69 cache.emplace_back(cmd);
70 }
71 }

74 hilet i = std::find(cache.cbegin(), cache.cend(), cmd);
75 if (i != cache.cend()) {
76 cache.erase(i);

78 }
