14color_primaries_to_RGBtoXYZ(float wx, float wy, float rx, float ry, float gx, float gy, float bx, float by) noexcept
15{
16 hilet w = vector3{wx, wy, 1.0f - wx - wy};
17 hilet r = vector3{rx, ry, 1.0f - rx - ry};
18 hilet g = vector3{gx, gy, 1.0f - gx - gy};
19 hilet b = vector3{bx, by, 1.0f - bx - by};
20
21 // Calculate white point's tristimulus values from coordinates
22 hilet W = vector3{1.0f * (w.x() / w.y()), 1.0f, 1.0f * (w.z() / w.y())};
23
24 // C is the chromaticity matrix.
25 hilet C = matrix3{r, g, b};
26
27 // solve tristimulus sums.
28 hilet S = scale3{~C * W};
29
30 return C * S;
31}
