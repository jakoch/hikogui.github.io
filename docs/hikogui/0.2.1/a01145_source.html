55 [[nodiscard]] bool handle_event(command command) noexcept
56 {
57 ttlet lock = std::scoped_lock(gui_system_mutex);
58
59 if (*enabled) {
60 switch (command) {
61 case command::gui_activate: this->_notifier(); return true;
62 case command::gui_enter:
63 this->_notifier();
64 this->window.update_keyboard_target(
65 this->shared_from_this(), keyboard_focus_group::normal, keyboard_focus_direction::forward);
66 return true;
67 default:;
68 }
69 }
70
71 return super::handle_event(command);
72 }

74 [[nodiscard]] bool handle_event(mouse_event const &event) noexcept final
75 {
76 ttlet lock = std::scoped_lock(gui_system_mutex);
77 auto handled = super::handle_event(event);
78
79 if (event.cause.leftButton) {
80 handled = true;
81 if (*enabled) {
82 if (compare_then_assign(_pressed, static_cast<bool>(event.down.leftButton))) {
83 window.request_redraw(window_clipping_rectangle());
84 }
85
86 if (event.type == mouse_event::Type::ButtonUp && _window_rectangle.contains(event.position)) {
87 handled |= handle_event(command::gui_activate);
88 }
89 }
90 }
91 return handled;
92 }

94 [[nodiscard]] hit_box hitbox_test(f32x4 window_position) const noexcept final
95 {
96 ttlet lock = std::scoped_lock(gui_system_mutex);
97
98 if (_window_clipping_rectangle.contains(window_position)) {
99 return hit_box{weak_from_this(), _draw_layer, *enabled ? hit_box::Type::Button : hit_box::Type::Default};
100 } else {
101 return hit_box{};
102 }
103 }
