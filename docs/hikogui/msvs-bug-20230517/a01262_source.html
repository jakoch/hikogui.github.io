50[[nodiscard]] inline float sRGB_linear_to_gamma(float u) noexcept
51{
52 if (u <= 0.0031308) {
53 return 12.92f * u;
54 } else {
55 return 1.055f * std::pow(u, 0.416f) - 0.055f;

170[[nodiscard]] inline color color_from_sRGB(std::string_view str)
171{
172 auto tmp = std::string{str};
173
174 if (tmp.starts_with("#")) {
175 tmp = tmp.substr(1);
176 }
177 if (tmp.size() != 6 && tmp.size() != 8) {
178 throw parse_error(std::format("Expecting 6 or 8 hex-digit sRGB color string, got {}.", str));
179 }
180 if (tmp.size() == 6) {
181 tmp += "ff";
182 }
183
184 hilet packed = from_string<uint32_t>(tmp, 16);
185
186 hilet r = truncate<uint8_t>(packed >> 24);
187 hilet g = truncate<uint8_t>(packed >> 16);
188 hilet b = truncate<uint8_t>(packed >> 8);
189 hilet a = truncate<uint8_t>(packed);
190 return color_from_sRGB(r, g, b, a);
191}
