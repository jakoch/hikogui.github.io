86[[nodiscard]] constexpr uint64_t operator_to_int(char const *s) noexcept
87{
88 uint64_t r = 0;
89 for (; *s != '\0'; s++) {
90 r <<= 5;
91 r |= static_cast<uint64_t>(char_to_graphic_character(*s));
92 }
93 return r;
94}
