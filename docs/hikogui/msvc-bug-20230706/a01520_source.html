62 constexpr auto shift = I * 5;
63 hilet x = (rhs._v >> shift) & 0x1f;
64 if (x == 0) {
65 return 0;
66 } else if (x <= 26) {
67 return 'a' + narrow_cast<char>(x - 1);
68 } else {
