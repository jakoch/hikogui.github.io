675template<typename T, byte_like Byte>
676[[nodiscard]] copy_cv_t<T, Byte>& implicit_cast(std::span<Byte> bytes)
677{
678 using value_type = copy_cv_t<T, Byte>;
679
680 static_assert(std::is_trivially_default_constructible_v<value_type>);
681 static_assert(std::is_trivially_destructible_v<value_type>);
682
683 if (sizeof(value_type) > bytes.size()) {
684 throw std::bad_cast();
685 }
686 hi_axiom_not_null(bytes.data());
687
688 if constexpr (alignof(value_type) != 1) {
689 if (std::bit_cast<std::uintptr_t>(bytes.data()) % alignof(value_type) != 0) {
690 throw std::bad_cast();
691 }
692 }
693
694 return *reinterpret_cast<value_type *>(bytes.data());
