18[[nodiscard]] inline std::byte *locked_memory_allocator_allocate(std::size_t n) noexcept
19{
20 auto p = VirtualAlloc(NULL, n, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
21 if (p == NULL) {
22 hi_log_fatal("Could not allocate locked memory. '{}'", get_last_error_message());
23 }
24
25 auto handle = GetCurrentProcess();
26 SIZE_T minimum_working_set_size;
27 SIZE_T maximum_working_set_size;
28
29 if (!GetProcessWorkingSetSize(handle, &minimum_working_set_size, &maximum_working_set_size)) {
30 hi_log_warning("Could not get process working set size. '{}'", get_last_error_message());
31
32 } else {
33 minimum_working_set_size += ceil(n, SIZE_T{4096});
34 maximum_working_set_size += ceil(n, SIZE_T{4096});
35 if (!SetProcessWorkingSetSize(handle, minimum_working_set_size, maximum_working_set_size)) {
36 hi_log_warning(
37 "Could not set process working set size to {}:{}. '{}'",
38 minimum_working_set_size,
39 maximum_working_set_size,
40 get_last_error_message());
41
42 } else {
43 if (!VirtualLock(p, n)) {
44 hi_log_warning("Could not lock memory. '{}'", get_last_error_message());
45 }
46 }
47 }
48 return static_cast<std::byte *>(p);
49}

51inline void locked_memory_allocator_deallocate(std::byte *p, std::size_t n) noexcept
52{
53 if (not VirtualUnlock(p, n)) {
54 hi_log_warning("Could not unlock memory. '{}'", get_last_error_message());
55
56 } else {
57 auto handle = GetCurrentProcess();
58 SIZE_T minimum_working_set_size;
59 SIZE_T maximum_working_set_size;
60
61 if (!GetProcessWorkingSetSize(handle, &minimum_working_set_size, &maximum_working_set_size)) {
62 hi_log_warning("Could not get process working set size. '{}'", get_last_error_message());
63
64 } else {
65 minimum_working_set_size -= ceil(n, SIZE_T{4096});
66 maximum_working_set_size -= ceil(n, SIZE_T{4096});
67 if (!SetProcessWorkingSetSize(handle, minimum_working_set_size, maximum_working_set_size)) {
68 hi_log_warning(
69 "Could not set process working set size to {}:{}. '{}'",
70 minimum_working_set_size,
71 maximum_working_set_size,
72 get_last_error_message());
73 }
74 }
75 }
76
77 if (not VirtualFree(p, 0, MEM_RELEASE)) {
78 hi_log_fatal("Could not deallocate locked memory. '{}'", get_last_error_message());
79 }
80}
