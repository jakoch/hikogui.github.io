107 font& register_font_file(std::filesystem::path const& path, bool post_process = true)
108 {
109 auto font = std::make_unique<true_type_font>(path);
110 auto font_ptr = font.get();
111
112 hi_log_info("Parsed font {}: {}", path.string(), to_string(*font));
113
114 hilet font_family_id = register_family(font->family_name);
115 _font_variants[*font_family_id][font->font_variant()] = font_ptr;
116
117 _fonts.emplace_back(std::move(font));
118 _font_ptrs.push_back(font_ptr);
119
120 if (post_process) {
121 this->post_process();
122 }
123
124 return *font_ptr;

154 void post_process() noexcept
155 {
156 // Sort the list of fonts based on the amount of unicode code points it supports.
157 std::sort(begin(_font_ptrs), end(_font_ptrs), [](hilet& lhs, hilet& rhs) {
158 return lhs->char_map.count() > rhs->char_map.count();
159 });
160
161 hilet regular_fallback_chain = make_fallback_chain(font_weight::regular, font_style::normal);
162 hilet bold_fallback_chain = make_fallback_chain(font_weight::bold, font_style::normal);
163 hilet italic_fallback_chain = make_fallback_chain(font_weight::regular, font_style::italic);
164
165 hi_log_info(
166 "Post processing fonts number={}, regular-fallback={}, bold-fallback={}, italic-fallback={}",
167 size(_fonts),
168 size(regular_fallback_chain),
169 size(bold_fallback_chain),
170 size(italic_fallback_chain));
171
172 // For each font, find fallback list.
173 for (hilet& font : _font_ptrs) {
174 auto fallback_chain = std::vector<hi::font *>{};
175
176 // Put the fonts from the same family, italic and weight first.
177 for (hilet& fallback : _font_ptrs) {
178 // clang-format off
179 if (
180 (fallback != font) and
181 (fallback->family_name == font->family_name) and
182 (fallback->style == font->style) and
183 almost_equal(fallback->weight, font->weight)
184 ) {
185 fallback_chain.push_back(fallback);
186 }
187 // clang-format on
188 }
189
190 if (almost_equal(font->weight, font_weight::bold)) {
191 std::copy(begin(bold_fallback_chain), end(bold_fallback_chain), std::back_inserter(fallback_chain));
192 } else if (font->style == font_style::italic) {
193 std::copy(begin(italic_fallback_chain), end(italic_fallback_chain), std::back_inserter(fallback_chain));
194 } else {
195 std::copy(begin(regular_fallback_chain), end(regular_fallback_chain), std::back_inserter(fallback_chain));
196 }
197
198 font->fallback_chain = std::move(fallback_chain);
199 }
200 }

218 [[nodiscard]] font_family_id register_family(std::string_view family_name) noexcept
219 {
220 auto name = to_lower(family_name);
221
222 auto it = _family_names.find(name);
223 if (it == _family_names.end()) {
224 hilet family_id = font_family_id(_font_variants.size());
225 _font_variants.emplace_back();
226 _family_names[name] = family_id;
227 return family_id;
228 } else {
229 return it->second;
230 }

240 [[nodiscard]] font const& find_font(font_family_id family_id, font_variant variant) const noexcept
241 {
242 hi_assert(family_id);
243 hi_assert_bounds(*family_id, _font_variants);
244
245 hilet& variants = _font_variants[*family_id];
246 for (auto alternative_variant : alternatives(variant)) {
247 if (auto font = variants[alternative_variant]) {
248 return *font;
249 }
250 }
251
252 // If a family exists, there must be at least one font variant available.
253 hi_no_default();
254 }

281 [[nodiscard]] font_glyphs_type find_glyph(font const& font, hi::grapheme grapheme) const noexcept
282 {
283 // First try the selected font.
284 if (hilet glyph_ids = font.find_glyph(grapheme); not glyph_ids.empty()) {
285 return {font, std::move(glyph_ids)};
286 }
287
288 // Scan fonts which are fallback to this.
289 for (hilet fallback : font.fallback_chain) {
290 hi_axiom_not_null(fallback);
291 if (hilet glyph_ids = fallback->find_glyph(grapheme); not glyph_ids.empty()) {
292 return {*fallback, std::move(glyph_ids)};
293 }
294 }
295
296 // If all everything has failed, use the tofu block of the original font.
297 return {font, {glyph_id{0}}};
298 }

308 [[nodiscard]] font_glyph_type find_glyph(font const& font, char32_t code_point) const noexcept
309 {
310 // First try the selected font.
311 if (hilet glyph_id = font.find_glyph(code_point)) {
312 return {font, glyph_id};
313 }
314
315 // Scan fonts which are fallback to this.
316 for (hilet fallback : font.fallback_chain) {
317 hi_axiom_not_null(fallback);
318 if (hilet glyph_id = fallback->find_glyph(code_point)) {
319 return {*fallback, glyph_id};
320 }
321 }
322
323 // If all everything has failed, use the tofu block of the original font.
324 return {font, glyph_id{0}};
325 }

339 [[nodiscard]] std::vector<hi::font *> make_fallback_chain(font_weight weight, font_style style) noexcept
340 {
341 auto r = _font_ptrs;
342
343 std::stable_partition(begin(r), end(r), [weight, style](hilet& item) {
344 return (item->style == style) and almost_equal(item->weight, weight);
345 });
346
347 auto char_mask = std::bitset<0x11'0000>{};
348 for (auto& font : r) {
349 if (font->char_map.update_mask(char_mask) == 0) {
350 // This font did not add any code points.
351 font = nullptr;
352 }
353 }
354
355 std::erase(r, nullptr);
356 return r;
357 }
