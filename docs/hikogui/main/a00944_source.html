78template<audio_device_filter FirstFilter, audio_device_filter... Filters>
79[[nodiscard]] inline bool match_audio_device(audio_device const &device, FirstFilter &&first_filter, Filters &&...filters) noexcept
80{
81 if constexpr (std::same_as<FirstFilter, audio_device_state>) {
82 if (device.state() != first_filter) {
83 return false;
84 }
85 } else if constexpr (std::same_as<FirstFilter, audio_direction>) {
86 if (not to_bool(device.direction() & first_filter)) {
87 return false;
88 }
89 } else {
90 hi_static_no_default();
91 }
92
93 return match_audio_device(device, std::forward<Filters>(filters)...);
94}
