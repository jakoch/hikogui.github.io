125 template<draw_attribute T>
126 constexpr void add(T const& attribute) noexcept
127 {
128 if constexpr (std::is_same_v<T, quad_color>) {
129 if (num_colors++ == 0) {
130 fill_color = attribute;
131 } else {
132 line_color = attribute;
133 }
134 hi_axiom(num_colors <= 2);
135
136 } else if constexpr (std::is_same_v<T, color>) {
137 if (num_colors++ == 0) {
138 fill_color = quad_color{attribute};
139 } else {
140 line_color = quad_color{attribute};
141 }
142 hi_axiom(num_colors <= 2);
143
144 } else if constexpr (std::is_same_v<T, line_end_cap>) {
145 if (num_line_caps++ == 0) {
146 begin_line_cap = attribute;
147 end_line_cap = attribute;
148 } else {
149 end_line_cap = attribute;
150 }
151 hi_axiom(num_line_caps <= 2);
152
153 } else if constexpr (std::is_same_v<T, hi::border_side>) {
154 border_side = attribute;
155#ifndef NDEBUG
156 hi_assert(not _has_border_side);
157 _has_border_side = true;
158#endif
159
160 } else if constexpr (std::is_same_v<T, corner_radii>) {
161 corner_radius = attribute;
162#ifndef NDEBUG
163 hi_assert(not _has_corner_radii);
164 _has_corner_radii = true;
165#endif
166
167 } else if constexpr (std::is_same_v<T, aarectangle>) {
168 clipping_rectangle = attribute;
169#ifndef NDEBUG
170 hi_assert(not _has_clipping_rectangle);
171 _has_clipping_rectangle = true;
172#endif
173
174 } else if constexpr (std::is_same_v<T, float> or std::is_same_v<T, int>) {
175 line_width = narrow_cast<float>(attribute);
176#ifndef NDEBUG
177 hi_assert(not _has_line_width);
178 _has_line_width = true;
179#endif
180 } else {
181 hi_static_no_default();
182 }
183 }

186 constexpr void add(First const& first, Second const& second, Rest const&...rest) noexcept

630 [[nodiscard]] constexpr static rectangle
631 make_rectangle(line_segment const& line, float width, line_end_cap c1, line_end_cap c2) noexcept
632 {
633 auto right = line.direction();
634
635 hilet radius = width * 0.5f;
636 hilet n = normal(right, 0.0f);
637 hilet up = n * width;
638 hilet t = normalize(right);
639
640 auto origin = line.origin() - n * radius;
641
642 // Extend the line by the radius for rounded end-caps.
643 hilet radius_offset = t * radius;
644 if (c1 == line_end_cap::round) {
645 origin -= radius_offset;
646 right += radius_offset;
647 }
648 if (c2 == line_end_cap::round) {
649 right += radius_offset;
650 }
651
652 return rectangle{origin, right, up};
653 }
654
655 [[nodiscard]] constexpr static rectangle make_rectangle(hi::circle const& circle) noexcept
656 {
657 hilet circle_ = f32x4{circle};
658 hilet origin = point3{circle_.xyz1() - circle_.ww00()};
659 hilet right = vector3{circle_.w000() * 2.0f};
660 hilet up = vector3{circle_._0w00() * 2.0f};
661 return rectangle{origin, right, up};
662 }
663
664 [[nodiscard]] constexpr static corner_radii make_corner_radii(float width, line_end_cap c1, line_end_cap c2) noexcept
665 {
666 auto r = f32x4::broadcast(width * 0.5f);
667
668 if (c1 == line_end_cap::flat) {
669 r = set_zero<0b0101>(r);
670 }
671 if (c2 == line_end_cap::flat) {
672 r = set_zero<0b1010>(r);
673 }
674
675 return corner_radii{r};
676 }
677
678 [[nodiscard]] constexpr static corner_radii make_corner_radii(hi::circle const& circle) noexcept
679 {
680 return corner_radii{f32x4{circle}.wwww()};
681 }
