18// B(t)=(P_{2}-P_{1})t+P_{1}
19hi_export template<typename T>
20constexpr std::array<T, 2> bezierToPolynomial(T P1, T P2) noexcept
21{
22 return {P2 - P1, P1};

25// B(t)=(P_{1}-2C+P_{2})t^{2}+2(C-P_{1})t+P_{1}
26template<typename T>
27hi_export constexpr std::array<T, 3> bezierToPolynomial(T P1, T C, T P2) noexcept
28{
29 return {P1 - C * 2 + P2, (C - P1) * 2, P1};
30}
31
32// B(t)=(-P_{1}+3C_{1}-3C_{2}+P_{2})t^{3}+(3P_{1}-6_{1}+3C_{2})t^{2}+(-3P_{1}+3C_{1})t+P_{1}
33template<typename T>
34hi_export constexpr std::array<T, 4> bezierToPolynomial(T P1, T C1, T C2, T P2) noexcept
35{
36 return {-P1 + C1 * 3 - C2 * 3 + P2, P1 * 3 - C1 * 6 + C2 * 3, P1 * -3 + C1 * 3, P1};
37}

104hi_export inline lean_vector<float> bezierFindTForNormalsIntersectingPoint(point2 P1, point2 P2, point2 P) noexcept
105{
106 hilet t_above = dot(P - P1, P2 - P1);
107 hilet t_below = dot(P2 - P1, P2 - P1);
108 if (t_below == 0.0) {
109 [[unlikely]] return {};
110 } else {
111 return {t_above / t_below};
112 }

119hi_export inline lean_vector<float> bezierFindTForNormalsIntersectingPoint(point2 P1, point2 C, point2 P2, point2 P) noexcept
120{
121 hilet P1_ = static_cast<f32x4>(P1);
122 hilet P2_ = static_cast<f32x4>(P2);
123 hilet C_ = static_cast<f32x4>(C);
124
125 hilet p = P - P1;
126 hilet p1 = C - P1;
127 hilet p2 = vector2{P2_ - (2 * C_) + P1_};
128
129 hilet a = dot(p2, p2);
130 hilet b = 3 * dot(p1, p2);
131 hilet c = dot(2 * p1, p1) - dot(p2, p);
132 hilet d = -dot(p1, p);
133 return solvePolynomial(a, b, c, d);
134}

143hi_export inline lean_vector<float> bezierFindX(point2 P1, point2 P2, float y) noexcept
144{
145 if (y < std::min({P1.y(), P2.y()}) || y > std::max({P1.y(), P2.y()})) {
146 return {};
147 }
148
149 auto r = lean_vector<float>{};
150 for (hilet t : bezierFindT(P1.y(), P2.y(), y)) {
151 if (t >= 0.0f && t < 1.0f) {
152 r.push_back(bezierPointAt(P1, P2, t).x());
153 }
154 }
155
156 return r;

218hi_export inline float bezierFlatness(point2 P1, point2 C, point2 P2) noexcept
219{
220 hilet P1P2 = hypot(P2 - P1);
221 if (P1P2 == 0.0f) {
222 return 1.0;
223 }
224
225 hilet P1C1 = hypot(C - P1);
226 hilet C1P2 = hypot(P2 - C);
227 return P1P2 / (P1C1 + C1P2);
228}

255hi_export inline std::optional<point2> getIntersectionPoint(point2 A1, point2 A2, point2 B1, point2 B2) noexcept
256{
257 // convert points to vectors.
258 hilet p = A1;
259 hilet r = A2 - A1;
260 hilet q = B1;
261 hilet s = B2 - B1;
262
263 // find t and u in:
264 // p + t*r == q + us
265 hilet crossRS = cross(r, s);
266 if (crossRS == 0.0f) {
267 // Parallel, other non, or a range of points intersect.
268 return {};
269
270 } else {
271 hilet q_min_p = q - p;
272 hilet t = cross(q_min_p, s) / crossRS;
273 hilet u = cross(q_min_p, r) / crossRS;
274
275 if (t >= 0.0f && t <= 1.0f && u >= 0.0f && u <= 1.0f) {
276 return bezierPointAt(A1, A2, t);
277 } else {
278 // The lines intersect outside of one or both of the segments.
279 return {};
280 }
281 }
282}
