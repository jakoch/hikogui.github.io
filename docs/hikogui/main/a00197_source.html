137 {
138 if (auto ptr = _main.load(std::memory_order::acquire)) {

354private:
355 static loop *timer_init() noexcept
356 {
357 hi_assert(not _timer_thread.joinable());
358
359 _timer_thread = std::jthread{[](std::stop_token stop_token) {
360 _timer.store(std::addressof(loop::local()), std::memory_order::release);
361
362 set_thread_name("timer");
363 loop::local().resume(stop_token);
364 }};
365
366 while (true) {
367 if (auto ptr = _timer.load(std::memory_order::relaxed)) {
368 return ptr;
369 }
370 std::this_thread::sleep_for(std::chrono::milliseconds(20));
371 }
372 }
373
374 static void timer_deinit() noexcept
375 {
376 if (auto const *const ptr = _timer.exchange(nullptr, std::memory_order::acquire)) {
377 hi_assert(_timer_thread.joinable());
378 _timer_thread.request_stop();
379 _timer_thread.join();
380 }
381 }
