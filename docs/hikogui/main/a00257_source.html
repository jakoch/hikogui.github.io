54template<typename T>
55hi_force_inline constexpr lean_vector<T> solvePolynomial(T const& a, T const& b, T const& c) noexcept
56{
57 if (a == 0) {
58 return solvePolynomial(b, c);
59 } else {
60 hilet D = b * b - static_cast<T>(4) * a * c;
61 if (D < 0) {
62 return {};
63 } else if (D == 0) {
64 return {-b / (static_cast<T>(2) * a)};
65 } else {
66 hilet Dsqrt = sqrt(D);
67 return {(-b - Dsqrt) / (static_cast<T>(2) * a), (-b + Dsqrt) / (static_cast<T>(2) * a)};
68 }
69 }
70}

74template<typename T>
75hi_force_inline lean_vector<T> solveDepressedCubicTrig(T const& p, T const& q) noexcept
76{
77 constexpr T oneThird = static_cast<T>(1) / static_cast<T>(3);
78 constexpr T pi2_3 = (static_cast<T>(2) / static_cast<T>(3)) * std::numbers::pi_v<T>;
79 constexpr T pi4_3 = (static_cast<T>(4) / static_cast<T>(3)) * std::numbers::pi_v<T>;
80
81 hilet U = oneThird * acos(((static_cast<T>(3) * q) / (static_cast<T>(2) * p)) * sqrt(static_cast<T>(-3) / p));
82 hilet V = static_cast<T>(2) * sqrt(-oneThird * p);
83
84 hilet t0 = V * cos(U);
85 hilet t1 = V * cos(U - pi2_3);
86 hilet t2 = V * cos(U - pi4_3);
87 return {t0, t1, t2};
88}

90template<typename T>
91hi_force_inline lean_vector<T> solveDepressedCubicCardano(T const& p, T const& q, T const& D) noexcept
92{
93 hilet sqrtD = sqrt(D);
94 hilet minusHalfQ = static_cast<T>(-0.5) * q;
95 hilet v = cbrt(minusHalfQ + sqrtD);
96 hilet w = cbrt(minusHalfQ - sqrtD);
97 return {v + w};

116template<typename T>
117hi_force_inline lean_vector<T> solveDepressedCubic(T const& p, T const& q) noexcept
118{
119 constexpr T oneForth = static_cast<T>(1) / static_cast<T>(4);
120 constexpr T oneTwentySeventh = static_cast<T>(1) / static_cast<T>(27);
121
122 if (p == 0.0 && q == 0.0) {
123 return {static_cast<T>(0)};
124
125 } else {
126 hilet D = oneForth * q * q + oneTwentySeventh * p * p * p;
127
128 if (D < 0 && p != 0.0) {
129 // Has three real roots.
130 return solveDepressedCubicTrig(p, q);
131
132 } else if (D == 0 && p != 0.0) {
133 // Has two real roots, or maybe one root
134 hilet t0 = (static_cast<T>(3) * q) / p;
135 hilet t1 = (static_cast<T>(-3) * q) / (static_cast<T>(2) * p);
136 return {t0, t1, t1};
137
138 } else {
139 // Has one real root.
140 return solveDepressedCubicCardano(p, q, D);
141 }
142 }
143}

153template<typename T>
154hi_force_inline constexpr lean_vector<T> solvePolynomial(T const& a, T const& b, T const& c, T const& d) noexcept
155{
156 if (a == 0) {
157 return solvePolynomial(b, c, d);
158
159 } else {
160 hilet p = (static_cast<T>(3) * a * c - b * b) / (static_cast<T>(3) * a * a);
161 hilet q = (static_cast<T>(2) * b * b * b - static_cast<T>(9) * a * b * c + static_cast<T>(27) * a * a * d) /
162 (static_cast<T>(27) * a * a * a);
163
164 hilet b_3a = b / (static_cast<T>(3) * a);
165
166 auto r = solveDepressedCubic(p, q);
167 for (auto &x: r) {
168 x -= b_3a;
169 }
170
171 return r;
172 }
173}
