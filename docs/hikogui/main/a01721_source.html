55[[nodiscard]] inline std::size_t get_bits(std::span<std::byte const> buffer, std::size_t &index, std::size_t length) noexcept
56{
57 hi_axiom(length <= sizeof(std::size_t) * CHAR_BIT);
58
59 auto value = 0;
60
61 auto todo = length;
62 auto done = 0_uz;
63 while (todo) {
64 hilet byte_index = index >> 3;
65 hilet bit_index = index & 7;
66 hi_axiom(byte_index < buffer.size());
67
68 hilet available_bits = 8 - bit_index;

72
73 hilet tmp = static_cast<int>(buffer[byte_index] >> bit_index) & mask;
74 value |= tmp << done;
75
76 todo -= nr_bits;
77 done += nr_bits;
78 index += nr_bits;
79 }
