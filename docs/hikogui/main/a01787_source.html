116hi_export [[nodiscard]] inline std::optional<std::filesystem::path> source_dir() noexcept
117{
118 using namespace std::literals;
119
120 // If the cmake_install.cmake file exists then the executable is located in a build directory.
121 hilet cmake_install_path = executable_dir() / "cmake_install.cmake";
122
123 if (std::filesystem::exists(cmake_install_path)) {
124 auto line = std::string{};
125 try {
126 auto fd = std::ifstream{cmake_install_path.string()};
127 line = getline(fd, 512);
128 fd.close();
129
130 } catch (...) {
131 return std::nullopt;
132 }
133
134 hilet cmake_install_start = "# Install script for directory: "s;
135 if (not line.starts_with(cmake_install_start)) {
136 return std::nullopt;
137 }
138
139 auto source_dir = std::filesystem::path{line.substr(cmake_install_start.size())};
140 if (not std::filesystem::exists(source_dir)) {
141 return std::nullopt;
142 }
143
144 return source_dir;
145
146 } else {
147 return std::nullopt;
148 }
149}

157[[nodiscard]] inline std::optional<std::filesystem::path> library_install_dir() noexcept
158{
159 // path is:
160 // - /install_dir/include/hikogui/path/path_location_impl.hpp
161 // - /build_dir/src/hikogui/path/path_location_impl.hpp
162 // becomes:
163 // - /install_dir/
164 // - /build_dir/
165 auto path = std::filesystem::path{__FILE__};
166 path.replace_filename("../../..");
167 auto install_path = std::filesystem::canonical(path);
168
169 if (install_path != library_source_dir()) {
170 return install_path;
171 } else {
172 return std::nullopt;
173 }
174}
