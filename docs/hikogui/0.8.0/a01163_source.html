122[[nodiscard]] inline std::size_t advance_thread_affinity(std::size_t &cpu) noexcept
123{
124 auto available_cpus = process_affinity_mask();
125 hi_assert_bounds(cpu, available_cpus);
126
127 ssize_t selected_cpu = -1;
128 do {
129 if (available_cpus[cpu]) {
130 try {
131 set_thread_affinity(cpu);
132 selected_cpu = narrow_cast<ssize_t>(cpu);
133 } catch (os_error const &) {
134 }
135 }
136
137 // Advance to the next available cpu.
138 // We do this, so that the caller of this function can detect a wrap around.
139 do {
140 if (++cpu >= available_cpus.size()) {
141 cpu = 0;
142 }
143 } while (!available_cpus[cpu]);
144 } while (selected_cpu < 0);
145
146 return narrow_cast<std::size_t>(selected_cpu);
147}
