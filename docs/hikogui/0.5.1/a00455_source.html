173 float baselineOffset(float height) noexcept {
174 if (alignment == vertical_alignment::top) {
175 return height - topAccender();
176 } else if (alignment == vertical_alignment::bottom) {
177 return bottomDescender();
178 } else if (alignment == vertical_alignment::middle) {
179 return height * 0.5f - middleCapHeight() * 0.5f;
180 } else {
181 tt_no_default();
182 }
183 }

190 float middleOffset(float height) const noexcept {
191 if (alignment == vertical_alignment::top) {
192 return height - topCapHeight() * 0.5f;
193 } else if (alignment == vertical_alignment::bottom) {
194 return height - bottomCapHeight() * 0.5f;
195 } else if (alignment == vertical_alignment::middle) {
196 return height - middleCapHeight() * 0.5f;
197 } else {
198 tt_no_default();
