38template<global_state_type Level, fixed_string SourcePath, int SourceLine, fixed_string Fmt, typename... Values>
39class log_message : public log_message_base {
40public:
41 static_assert(std::popcount(to_underlying(Level)) == 1);
42
43 // clang-format off
44 static constexpr char const *log_level_name =
45 Level == global_state_type::log_fatal ? "fatal" :
46 Level == global_state_type::log_error ? "error" :
47 Level == global_state_type::log_warning ? "warning" :
48 Level == global_state_type::log_info ? "info" :
49 Level == global_state_type::log_debug ? "debug" :
50 Level == global_state_type::log_trace ? "trace" :
51 Level == global_state_type::log_audit ? "audit" :
52 Level == global_state_type::log_statistics ? "stats" :
53 "<unknown log level>";
54 // clang-format on
55
56 log_message(log_message const&) noexcept = default;
57 log_message& operator=(log_message const&) noexcept = default;
58
59 template<typename... Args>
60 hi_force_inline log_message(Args&&...args) noexcept :
61 _time_stamp(time_stamp_count::inplace_with_thread_id{}), _what(std::forward<Args>(args)...)
62 {
63 }
64
65 std::string format() const noexcept override
66 {
67 hilet utc_time_point = time_stamp_utc::make(_time_stamp);
68 hilet sys_time_point = std::chrono::clock_cast<std::chrono::system_clock>(utc_time_point);
69 hilet local_time_point = cached_current_zone().to_local(sys_time_point);
70
71 hilet cpu_id = _time_stamp.cpu_id();
72 hilet thread_id = _time_stamp.thread_id();
73 hilet thread_name = get_thread_name(thread_id);
74
75 if constexpr (to_bool(Level & global_state_type::log_statistics)) {
76 return std::format("{} {}({}) {:5} {}\n", local_time_point, thread_name, cpu_id, log_level_name, _what());
77 } else {
78 auto source_filename = std::filesystem::path{static_cast<std::string_view>(SourcePath)}.filename().generic_string();
79 return std::format(
80 "{} {}({}) {:5} {} ({}:{})\n",
81 local_time_point,
82 thread_name,
83 cpu_id,
84 log_level_name,
85 _what(),
86 source_filename,
87 SourceLine);
88 }
89 }
90
91 [[nodiscard]] std::unique_ptr<log_message_base> make_unique_copy() const noexcept override
92 {
93 return std::make_unique<log_message>(*this);
94 }
95
96private:
97 time_stamp_count _time_stamp;
98 delayed_format<Fmt, Values...> _what;
99};

112 template<global_state_type Level, fixed_string SourcePath, int SourceLine, fixed_string Fmt, typename... Args>
113 hi_force_inline void add(Args&&...args) noexcept
114 {
115 static_assert(std::popcount(to_underlying(Level)) == 1);
116
117 hilet state = global_state.load(std::memory_order::relaxed);
118 if (not to_bool(state & Level)) {
119 return;
120 }
121
122 // Add messages in the queue, block when full.
123 // * This reduces amount of instructions needed to be executed during logging.
124 // * Simplifies logged_fatal_message logic.
125 // * Will make sure everything gets logged.
126 // * Blocking is bad in a real time thread, so maybe count the number of times it is blocked.
127
128 // Emplace a message directly on the queue.
129 _fifo.emplace<detail::log_message<Level, SourcePath, SourceLine, Fmt, forward_value_t<Args>...>>(
130 std::forward<Args>(args)...);
131
132 if (to_bool(Level & global_state_type::log_fatal) or not to_bool(state & global_state_type::log_is_running)) {
133 // If the logger did not start we will log in degraded mode and log from the current thread.
134 // On fatal error we also want to log from the current thread.
135 [[unlikely]] flush();
136 }
137 }

143 hi_no_inline void flush() noexcept;
